y2k_df = (df[(df['Year'] == 2000) & ~df['Code'].isna()]
          .drop(columns='Region')
          .rename(columns={'Entity': 'Country Name'})
          .dropna()
          [['Code', 'Human Development Index', 'Fertility Rate', 'Population', 'Country Name']]
         )
